void OneTriangle(size_t height)
{
    for (int i = 0; i < height; i++)
    {
        for (int j = 0; j < height; j++)
        {
            if (i + j >= height - 1)
            {
                cout << "# ";
            }
            else
            {
                cout << "  ";
            }
        }
        cout << endl;
    }
}

void TwoTriangle(size_t height)
{
    for (int i = 0; i < height; i++)
    {
        for (int j = 0; j < height; j++)
        {
            if (i + j >= height - 1)
                cout << "#";
            else
                cout << " ";
        }
        cout << " ";
        for (int j = 0; j < height; j++)
        {
            if (i >= j)
                cout << "#";
            else
                cout << " ";
        }
        cout << endl;
    }
}
